{
  Variáveis externas disponíveis:
    PathXsd: string;
    MappingXsdFileName: string;
    Input: string;
    Datasets: TspdXSDDataset;
    Params : TStringList;
}
Var
  _Xml : string;
begin
  Progress('Carregando configurações de XSDDataset.');
  Datasets.XsdFile := PathXsd + 'servico_consultar_nfse_resposta.xsd';
  Datasets.XsdElementName := 'ConsultarNfseResposta';
  Datasets.ConfigSection := 'RETORNO_CONSULTARNFSE';
  if Trim(MappingXsdFileName) <> '' then
    Datasets.MappingFileName := MappingXsdFileName;

  Progress('Criando XSDDataset.');
  Datasets.CreateXSDDataset;

  _Xml := StringReplace(Input, '<![CDATA[<?xml version="1.0" encoding="UTF-8" standalone="yes"?>', '', []);
  _Xml := StringReplace(_Xml, ']]>', '', []);

  _Xml := XmlDeepNodeSearch(_Xml, Datasets.XsdElementName, '//*[namespace-uri()="http://tempuri.org/servico_consultar_nfse_resposta.xsd" and local-name()="ConsultarNfseResposta"]');

  if Pos('<faultstring',input) > 0 then
    Params.Add('faultstring='+input)
  else
  begin
    Datasets.LoadFromXMLContent(_Xml);
    _Xml := XmlSelect(_Xml, '//*[namespace-uri()="http://tempuri.org/servico_consultar_nfse_resposta.xsd" and local-name()="CompNfse"]');
    Params.Add('XmlNFSe='+_Xml);
    Progress('XML carregado com sucesso.');
  end;
end.
