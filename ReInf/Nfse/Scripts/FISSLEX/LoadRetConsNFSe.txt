{
  Variáveis externas disponíveis:
    PathXsd: string;
    MappingXsdFileName: string;
    Input: string;
    Datasets: TspdXSDDataset;
    Params : TStringList;
}

{$I ..\Comuns\Utils.txt}

Var
  _Xml : string;
begin
  Progress('Carregando configurações de XSDDataset.');
  Datasets.XsdFile := PathXsd + 'nfse.xsd';
  Datasets.XsdElementName := 'ConsultarNfseResposta';
  Datasets.ConfigSection := 'RETORNO_CONSULTARNFSE';
  if Trim(MappingXsdFileName) <> '' then
    Datasets.MappingFileName := MappingXsdFileName;

  Progress('Criando XSDDataset.');
  Datasets.CreateXSDDataset;

  _Xml := StringReplace(Input,'Consultarnfseresposta','ConsultarNfseResposta',[rfReplaceAll]);

  if Pos('&amp;', _Xml) = 0 then
    _Xml := StringReplace(_Xml, '&',  '&amp;',  [rfReplaceAll]);
  _Xml := CorrigeXml(_Xml);
  if Pos('<!--CRLF-->', _Xml) > 0 then
  begin
    _Xml := StringReplace(_Xml, '<!--CRLF-->', #13#10, [rfReplaceAll]);

    _Xml := XmlDeepNodeSearch(_Xml, Datasets.XsdElementName, '//*[namespace-uri()="FISS-LEX" and local-name()="ConsultarNfseResposta"]');

    _Xml := StringReplace(_Xml, #13#10, '<!--CRLF-->', [rfReplaceAll]);
  end
  else
    _Xml := XmlDeepNodeSearch(_Xml, Datasets.XsdElementName, '//*[namespace-uri()="FISS-LEX" and local-name()="ConsultarNfseResposta"]');


  if Pos('<faultstring',input) > 0 then
    Params.Add('faultstring='+input)
  else
  begin
    Datasets.LoadFromXMLContent(_Xml);
    _Xml := XmlSelect(_Xml, '//*[namespace-uri()="FISS-LEX" and local-name()="CompNfse"]');
    Params.Add('XmlNFSe='+_Xml);
    Progress('XML carregado com sucesso.');
  end;
end.
