{
  Vari?veis externas dispon?veis:
    Output: string;
    Datasets: TspdXSDDataset;
}
  function MontarXml: String;
  var
    _i, _j, _k: integer;
    _ds: TSpdXSDClientDataSet;
    _delimitadorCampo : String;
  begin
    _delimitadorCampo := '';
    Result := '';

    _ds := Datasets.GetDataSetByName('Xml');
    _ds.First;
    if _ds <> nil then
    begin
      for _i := 1 to _ds.Fields.Count - 1 do
      begin
        Result := Result + _ds.Fields[_i].AsString + _delimitadorCampo;
      end;
    end;
    
    _ds := Datasets.GetDataSetByName('Rps');
    _ds.First;
    while not _ds.Eof do
    begin
      if _ds <> nil then
      begin
        for _i := 2 to _ds.Fields.Count - 1 do
        begin
          Result := Result + _ds.Fields[_i].AsString + _delimitadorCampo;
        end;
      end;
      _ds.next;
    end;

    _ds := Datasets.GetDataSetByName('Rodape');
    _ds.First;
    while not _ds.Eof do
    begin
      if _ds <> nil then
      begin
        for _i := 2 to _ds.Fields.Count - 1 do
        begin
          Result := Result + _ds.Fields[_i].AsString + _delimitadorCampo;
        end;
      end;
      _ds.next;
    end;
    
    Result := StringReplace(Result, '§', ' ', [rfReplaceAll]);
  end;

begin
  Progress('Gerando arquivo XML.');
  Output := MontarXml;
  Progress('Arquivo XML gerado com sucesso.');
end.
